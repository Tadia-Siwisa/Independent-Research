        // ===== CINEMATIC SEQUENCE =====
        const seqLogo = document.getElementById('seqLogo');
        const seqMission = document.getElementById('seqMission');
        const seqIndustries = document.getElementById('seqIndustries');
        const openingSequence = document.getElementById('openingSequence');
        const mainSite = document.getElementById('mainSite');

        // Stage 1: Logo appears (0ms)
        setTimeout(() => {
            seqLogo.classList.add('active');
        }, 300);

        // Stage 2: Logo fades, mission appears (2400ms) - Adjusted to allow logo animation (1.8s) to complete
        setTimeout(() => {
            seqLogo.style.opacity = '0';
            seqLogo.style.transition = 'opacity 0.6s ease';
            seqMission.classList.add('active');
        }, 2400);

        // Stage 3: Mission fades, industries assemble (4400ms)
        setTimeout(() => {
            seqMission.style.opacity = '0';
            seqMission.style.transition = 'opacity 0.6s ease';
            seqIndustries.classList.add('active');
        }, 4400);

        // Industries expand one by one
        const industries = [
            document.getElementById('seqInd1'),
            document.getElementById('seqInd2'),
            document.getElementById('seqInd3'),
            document.getElementById('seqInd4'),
            document.getElementById('seqInd5')
        ];

        industries.forEach((ind, i) => {
            setTimeout(() => {
                ind.classList.add('open');
            }, 4800 + (i * 400));
        });

        // Sequence completes (7800ms)
        setTimeout(() => {
            openingSequence.classList.add('complete');
            mainSite.classList.add('visible');
        }, 7800);

        // ===== NAVBAR SCROLL =====
        const navbar = document.getElementById('navbar');
        window.addEventListener('scroll', () => {
            navbar.classList.toggle('scrolled', window.scrollY > 60);
        });

        // ===== SCROLL REVEAL =====
        const revealElements = document.querySelectorAll('.reveal');
        const revealObserver = new IntersectionObserver((entries) => {
            entries.forEach(entry => {
                if (entry.isIntersecting) {
                    entry.target.classList.add('visible');
                }
            });
        }, { threshold: 0.08, rootMargin: '0px 0px -40px 0px' });

        revealElements.forEach(el => revealObserver.observe(el));

        // ===== SMOOTH SCROLL =====
        document.querySelectorAll('a[href^="#"]').forEach(anchor => {
            anchor.addEventListener('click', function(e) {
                e.preventDefault();
                const target = document.querySelector(this.getAttribute('href'));
                if (target) {
                    target.scrollIntoView({ behavior: 'smooth', block: 'start' });
                }
            });
        });

        // ===== COMPANY CARD EXPAND =====
        function toggleCard(card) {
            const isExpanded = card.classList.contains('expanded');

            // Close all other cards
            document.querySelectorAll('.company-card.expanded').forEach(c => {
                if (c !== card) {
                    c.classList.remove('expanded');
                    c.setAttribute('aria-expanded', 'false');
                }
            });

            // Toggle current
            const newState = !isExpanded;
            card.classList.toggle('expanded', newState);
            card.setAttribute('aria-expanded', newState ? 'true' : 'false');
        }

        document.querySelectorAll('.company-card').forEach(card => {
            card.addEventListener('click', function(e) {
                // Don't expand if clicking the discuss button
                if (e.target.closest('.discuss-btn')) return;
                toggleCard(this);
            });

            card.addEventListener('keydown', function(e) {
                if (e.key === 'Enter' || e.key === ' ') {
                    e.preventDefault();
                    toggleCard(this);
                }
            });
        });
    </script>
